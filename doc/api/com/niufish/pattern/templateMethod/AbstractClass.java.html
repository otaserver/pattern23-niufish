public abstract class AbstractClass {
    
    /**
     * 模版方法<br />
     * 给出逻辑的骨架
     */
    public void templateMethod(){
        primitiveOperation1();
        primitiveOperation2();
    }

    /**
     * 基本操作1，子类须实现
     */
    protected abstract void primitiveOperation1();

    /**
     * 基本操作2，子类须实现
     */
    protected abstract void primitiveOperation2();

}
