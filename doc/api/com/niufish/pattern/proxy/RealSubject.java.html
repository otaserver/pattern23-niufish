public class RealSubject implements Subject {
    
    /**
     * 做这
     */
    private void doThis(){
        System.out.print("烧饼的价格：");
    }

    /**
     * 做那，反正干了不少事情
     * @return 成本
     */
    private int doThat(){
        return 2;
    }

    /**
     * @see com.niufish.pattern.proxy.Subject#request()
     */
    public int request() {
        doThis();
        return doThat();
    }

}
